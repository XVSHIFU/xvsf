import socket
import threading
import sys
 
class PortForwarder:
    def __init__(self, lh, lp, th, tp):
        self.lh = lh
        self.lp = lp
        self.th = th
        self.tp = tp
        self.running = False
 
    def forward(self, s, d, sn, dn):
        try:
            while self.running:
                data = s.recv(1024)
                if not data: break
                d.sendall(data)
        except:
            pass
        finally:
            s.close()
            d.close()
 
    def handle(self, cs):
        try:
            ts = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            ts.connect((self.th, self.tp))
            t1 = threading.Thread(target=self.forward, args=(cs, ts, "c", "t"))
            t2 = threading.Thread(target=self.forward, args=(ts, cs, "t", "c"))
            t1.daemon = True
            t2.daemon = True
            t1.start()
            t2.start()
            t1.join()
            t2.join()
        except:
            cs.close()
 
    def start(self):
        self.s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        try:
            self.s.bind((self.lh, self.lp))
            self.s.listen(5)
            self.running = True
            while self.running:
                cs, _ = self.s.accept()
                t = threading.Thread(target=self.handle, args=(cs,))
                t.daemon = True
                t.start()
        except:
            pass
        finally:
            self.running = False
            self.s.close()
 
if __name__ == "__main__":
    if len(sys.argv) != 5:
        print("Usage: python %s <listen_ip> <listen_port> <target_ip> <target_port>" % sys.argv[0])
        sys.exit(1)
    pf = PortForwarder(sys.argv[1], int(sys.argv[2]), sys.argv[3], int(sys.argv[4]))
    pf.start()
